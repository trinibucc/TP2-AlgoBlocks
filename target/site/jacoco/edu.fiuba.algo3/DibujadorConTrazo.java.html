package edu.fiuba.algo3;

import edu.fiuba.algo3.modelo.Posicion;
import javafx.scene.image.Image;

public class DibujadorConTrazo extends Dibujador {

    Image lineaHorizontal = new Image("h.jpg");
    Image lineaVertical = new Image("vertical.jpg");


    public DibujadorConTrazo(Posicion pos, Tablero tablero){
        super(pos, tablero);
    }



    protected void actualizar(){
        if(posicion.obtenerX() != inicio.obtenerX()){
            tablero.trazar(posicion);
        }
        else if (posicion.obtenerY() != inicio.obtenerY()){
            tablero.trazar(posicion);
            inicio = posicion;
        }
    }
}
