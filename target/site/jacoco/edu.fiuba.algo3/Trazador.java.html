package edu.fiuba.algo3;

import edu.fiuba.algo3.modelo.Posicion;
import edu.fiuba.algo3.modelo.SectorDibujo;
import edu.fiuba.algo3.modelo.Segmento;
import javafx.scene.layout.GridPane;
import javafx.scene.layout.Pane;

import javax.swing.*;
import java.util.List;
import java.util.Observable;
import java.util.Observer;

public abstract class Trazador extends Pane implements Observer {

    Tablero tablero;
    TrazadorHorizontal trazadorHorizontal;
    Posicion inicio = new Posicion(0,0);
    Posicion fin;
    TrazadorVertical trazadorVertical;

    public Trazador(Posicion posicion, Tablero tablero){
        this.fin = posicion;
        this.tablero = tablero;
        tablero.addObserver(this);
    }


    @Override
    public void update(Observable observable, Object o) {
        Trazador trazador = (Trazador) o;
        if (trazador.inicio.obtenerX() != trazador.fin.obtenerX()) {
            trazadorHorizontal.actualizar();
        }/* else
            trazadorVertical.actualizar();*/

    }


    protected abstract void actualizar();
}
