package edu.fiuba.algo3;

import edu.fiuba.algo3.modelo.Posicion;
import javafx.scene.canvas.Canvas;
import javafx.scene.canvas.GraphicsContext;
import javafx.scene.image.Image;
import javafx.scene.image.ImageView;

import java.util.Observable;

public class Tablero extends Observable {

    private Canvas tablero;

    public Tablero(Canvas tablero){
        this.tablero = tablero;
    }

    /*public ImageView obtenerImagen(Posicion posicion){
        return tablero[posicion.obtenerX()][posicion.obtenerY()];
    }*/

    public void inicializar(Dibujador t){
        t.actualizar();
        setChanged();
    }


    public void trazar(Posicion posicion) {
        GraphicsContext graphicsContext = tablero.getGraphicsContext2D();
        graphicsContext.lineTo(posicion.obtenerX(), posicion.obtenerY());
        graphicsContext.stroke();
    }
}
