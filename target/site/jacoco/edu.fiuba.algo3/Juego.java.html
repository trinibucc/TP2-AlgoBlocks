package edu.fiuba.algo3;

import edu.fiuba.algo3.modelo.AlgoBlocks;

import javafx.geometry.Insets;
import javafx.scene.control.Label;
import javafx.scene.image.Image;
import javafx.scene.image.ImageView;
import javafx.scene.layout.HBox;
import javafx.scene.layout.VBox;
import javafx.scene.paint.Color;

public class Juego {

    public Juego(){

    }

    public VBox crearVentana(){
        Personaje personaje = new Personaje();
        Algoritmo algoritmo = new Algoritmo(personaje);
        AlgoBlocks algoBlocks = new AlgoBlocks();
        algoritmo.agregarLista(algoBlocks);
        Image image = new Image("cuadrado-blanco.jpg");
        ImageView[][] tablero = new ImageView[10][10];

        CreadorSectorTablero creadorSectorTablero = new CreadorSectorTablero();
        Tablero juegoTablero = new Tablero(creadorSectorTablero);

        Titulo ventana = new Titulo();
        Label titulo = ventana.crearTitulo();

        BotonEjecutar ejecutar = new BotonEjecutar(algoritmo, juegoTablero);
        BotoneraMovimientos botoneraMovimientos = new BotoneraMovimientos(algoBlocks);
        BotonReiniciar reiniciar = new BotonReiniciar(algoBlocks, algoritmo);
        HBox central = new HBox(botoneraMovimientos, creadorSectorTablero, reiniciar);
        central.setSpacing(20);
        //central.setPadding(new Insets(20));
        VBox todo = new VBox(titulo, central, ejecutar);
        return todo;

    }
}
